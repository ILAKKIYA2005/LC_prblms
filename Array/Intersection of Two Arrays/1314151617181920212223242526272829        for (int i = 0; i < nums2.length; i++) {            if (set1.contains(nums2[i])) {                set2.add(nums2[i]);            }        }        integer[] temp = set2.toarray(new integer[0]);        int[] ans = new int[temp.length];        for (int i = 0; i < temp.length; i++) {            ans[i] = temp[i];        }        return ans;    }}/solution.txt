        for (int i = 0; i < nums2.length; i++) {
            if (set1.contains(nums2[i])) {
                set2.add(nums2[i]);
            }
        }

        Integer[] temp = set2.toArray(new Integer[0]);

        int[] ans = new int[temp.length];

        for (int i = 0; i < temp.length; i++) {
            ans[i] = temp[i];
        }

        return ans;
    }
